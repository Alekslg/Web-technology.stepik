git --version

git config --global user.name "Aleks"
git config --global user.email "aaa@sss.ew"
git config -l

git init .   # инициализация хранилища в текущей директории

git status
git add *  # or .              # Добавление изменений в новый снепшот
git commit -m "Comment"        # Создание снепшота с описанием

git log  # просмотр всех точек восстановлениея/снепшотов/коммитов
git log -2  #просмотр последний двух коммитов
git log -2 -p  # просмотр последний двух коммитов с доп. информацией

git checkout -- file   # отмена всех внесенных изменений после последнего коммита, до добавления в коммит
git diff --staged   # просмотр дополнительной информации о вносимых изменениях в коммит, перед его созданием

.gitignore   # перечень игнорируемых директорий и файлов

git clone url   # клонирование репозитория на локальную машину
git push origin  # Заливаем репозиторий в облако
git remote -v    # Просмотр ссылки для подключения к облаку репозитория
git remote set-url origin url   # Замена ссылки подключения к облаку репозитория
git.bash  ssh-keygen.exe   # Генерация ssh ключа в Windows
